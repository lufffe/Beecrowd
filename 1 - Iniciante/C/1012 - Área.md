    #include <stdio.h>

    int main() 
    {
       double a,b,c,pi;

       pi = 3.14159 ;
       scanf("%lf",&a);
       scanf("%lf",&b);
       scanf("%lf",&c);

       printf("TRIANGULO: %.3lf\n",((a*c)/2));
       printf("CIRCULO: %.3lf\n",(pi*pow(c,2)));
       printf("TRAPEZIO: %.3lf\n",((a+b)*c/2));
       printf("QUADRADO: %.3lf\n",(b*b));
       printf("RETANGULO: %.3lf\n",(a*b));

       return 0;
    }
