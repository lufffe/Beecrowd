>C
```C
#include <stdio.h>
	
int main()
{
	double pi,raio,A ;

	scanf("%lf",&raio);
	pi = 3.14159 ;
	A = (pi * pow(raio,2));
	
	printf("A=%.4lf\n",A);

	return 0;
}
```


>PYTHON 3.9
```Python 3.9
pi = 3.14159
raio = float(input())

A = (pi * (raio ** 2))

print(f"A={A:.4f}")
```
